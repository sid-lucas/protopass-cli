# Installation du module permettant de créer des environnements virtuels Python
sudo apt install python3.12-venv
# Installation du gestionnaire de paquets Python (pip)
sudo apt install python3-pip

cat >> requirements.txt <<EOF
bcrypt
pycryptodome
pyotp
argparse
requests
EOF

# bcrypt : pour le hachage du mot de passe maître
# pycryptodome : pour le chiffrement/déchiffrement des données (AES)
# pyotp : pour gérer les codes TOTP (authentification à 2 facteurs)
# argparse : pour créer une interface en ligne de commande (CLI)
# requests : pour faire des requêtes HTTP (utile pour l’API SimpleLogin)


# Création et activation d’un environnement virtuel
python3 -m venv .venv
source .venv/bin/activate
# Un environnement virtuel (venv) permet d’isoler les bibliothèques Python de ce projet des autres programmes du système. Les dépendances installées ici n’affectent pas le reste du système ni les autres projets. Le dossier .venv contient sa propre version de Python et de pip.


# Installation des dépendances dans l’environnement virtuel
pip install -r requirements.txt

# Project Python ≥ 3.3 only
